```bash
# 0. 取得連線字串
# 1. Instance Security 記得允許您的 IP 存取
# 2. 測試連線
$ psql -h [ENDPOINT_URL] \
       -U [USERNAME] \
       -d [DATABASE_NAME]
#     輸入密碼
#     檢查 - 列出所有資料庫
$ \l

#     檢查 - 列出所有 Tables
$ \d

# 3. 匯出
$ pg_dump -h [ENDPOINT_URL] \
          -U [USERNAME] \
          -f [FILENAME].sql [DATABASE_NAME]
```
